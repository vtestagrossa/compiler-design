function main a: integer, b: real, c: boolean returns real;
    d: integer is 8;
    e: boolean is not c;
    f: real is 1.9 + a;
begin
    if not (a < 0 or a > 0) and b /= 0 and e = false then
        reduce +
            a / b;
            b * a ** 2;
            f rem b;
        endreduce;
    else
        case a is
            when 1 => 1;
            when 2 => 2;
            when 3 =>
                if b <= 10 and b >= 1 then
                    b / a + f * b ** 2 - f rem b;
                else
                    a * d + d ** 2 - d rem a;
                endif;
            others => 111;
        endcase;
    endif;
end;
